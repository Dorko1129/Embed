print("This program calculates times table (only odd lines up to 13)")


tablenum = input("\nWhich multiplication table shall I generate for you? ")
tablenum = int(tablenum)

print("\nHere is your", tablenum, "times table (odd lines only up to 13):\n")

for i in range(1, 14):          # up to 13
    if i % 2 == 1:              # only odd numbers
        print(i, "times", tablenum, "is", i * tablenum)
        print("------------------")
